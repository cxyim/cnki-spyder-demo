    function InitOrgList(data) {
        var p = "";
        for (var i = 0; i < data.length; i++) {
            if (i <= 9) {
                var item = data[i];
                var FilterName = item.FilterName;
                var name = FilterName;
                if (FilterName && FilterName.length > 8) {
                    name = FilterName.autoCutStr(0, 8, true);
                }
                p += "<p class=\"jg-rr\"> .<a  name=\"aOrg\" style=\"cursor:pointer;\" title=\"" + FilterName + "\" target=\"_blank\" code=\"" + item.TypeCode + "\">" + name + "</a><p>";
            }

        }
        $("#rightOrg").html(p);
    }
    function RightSubjectInfo(content) {
